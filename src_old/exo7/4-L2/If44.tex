\contenu{
\texte{
On munit $E = \R_n[X]$ du produit scalaire :
Pour $P = \sum_i a_iX^i$ et $Q = \sum_i b_iX^i$,
$(P\mid Q) = \sum_i a_ib_i$.

Soit $H = \{ P \in E \text{ tq } P(1) = 0 \}$.
}
\begin{enumerate}
    \item \question{Trouver une base orthonormale de $H$.}
    \item \question{Calculer $d(X,H)$.}
\reponse{
$\frac1{\sqrt{n+1}}$.
}
\end{enumerate}
}
