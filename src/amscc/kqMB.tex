\contenu{
	\texte{
		On considère la fonction \( f \) définie sur \(\mathbb{R}\) par :
		\[
		f(y)= \begin{cases}C y^{3} & \text{si } 0 \leq y \leq 1 \\ 0 & \text{sinon.}\end{cases}
		\]
	}
	\begin{enumerate}
		\item \question{Pour quelle valeur de \( C \) la fonction \( f \) est-elle la densité d'une variable aléatoire \( Y \) ?}
		\item \question{Déterminer la fonction de répartition \( F \) de la variable aléatoire \( Y \).}
		\item \question{Calculer l'espérance de la variable aléatoire \( Y \).}
		\item \question{Déterminer la probabilité de l'événement \(\{0.4 < Y \leq 1.2\}\).}
	\end{enumerate}
}
