\contenu{
\texte{
Soient ${a,b} : \R \to \R$ continues, et $y,z$ solutions de
$\begin{cases}y(0) = z(0)\cr y' = a(t)y + b(t)\cr z'\le a(t)z + b(t).\end{cases}$

Démontrer que : $\forall\ t \ge 0$, on a $y(t) \ge z(t)$.
}
\reponse{
\'Etudier $e^{-A}(y-z)$, $A'=a$.
}
}
