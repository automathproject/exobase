\chapitre{Dérivabilité des fonctions réelles}

\niveau{L1}
\difficulte{}

\contenu{
\texte{
Soit $f:\R \to \R$ dérivable telle que $f'(x) \to \ell$ lorsque $x\to +\infty$.
Montrer que $\frac {f(x)}x \to \ell$ lorsque $x\to +\infty$.

Chercher un contrexemple pour la réciproque.
}
}
