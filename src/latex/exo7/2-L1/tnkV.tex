\contenu{
\texte{
Soit $f : {\R} \to  {\R}$ convexe. Montrer que l'on a :

-- soit $f$ est croissante sur $\R$.\par
-- soit $f$ est décroissante sur $\R$.\par
-- soit il existe $a \in \R$ tel que $f$ est décroissante sur
$]-\infty,a]$, puis croissante sur $[a,+\infty[$.
}
}
