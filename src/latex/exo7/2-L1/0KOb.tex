\contenu{
\texte{
Calculer $\sum_{n=0}^{+\infty}\sum_{k=n}^{+\infty}\frac1{k!}$.
}
\reponse{
$\sum_{n=0}^{+\infty}\sum_{k=n}^{+\infty}\frac1{k!}=
\sum_{k=0}^{+\infty}\frac{k+1}{k!}=2e$.
}
}
