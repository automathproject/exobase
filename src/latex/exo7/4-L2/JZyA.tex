\contenu{
\texte{
Soit $E = \mathcal{C}([0,1])$ muni du produit scalaire :
$(f\mid g) =  \int_{t=0}^1 fg(t)\,d t$, et $F = \{ f \in E \text{ tq } f(0) = 0 \}$.

Montrer que $F^\perp = \{0\}$.
}
\reponse{
$f \in F^\perp  \Rightarrow  xf\perp f$.
}
}
