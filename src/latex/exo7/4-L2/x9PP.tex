\texte{
Soit $G$ un sous-groupe de $(\Rr,+)$ avec  $G \neq \{0\}$.
}
\begin{enumerate}
    \item \question{Montrer l'existence de $\alpha =\inf (G\cap \Rr^{+*})$.}
    \item \question{Si $\alpha >0$ montrer que $G=\alpha \Zz.$}
    \item \question{Si $\alpha =0$ montrer que $G$ est dense dans $\Rr.$}
\end{enumerate}
