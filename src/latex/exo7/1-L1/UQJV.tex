\contenu{
\texte{
Soit $f\in \mathcal{L}(E)$ telle que $f^2 = 0$.
Montrer qu'il existe $g,h \in \mathcal{L}(E)$ tels que $f=g\circ h$ et $h\circ g = 0$.
}
\reponse{
$g =$ une projection sur $\Im f$ et $h=f$.
}
}
