\texte{
Soit $P \in {\R[X]}$. Existe-t-il $Q \in {\R[X]}$ tel que
$\forall\ \alpha \in \left]-\frac\pi2, \frac\pi2\right[,\
 P(\tan\alpha) = Q\Bigl(\frac 1{\cos\alpha}\Bigr)$ ?
}
\reponse{
oui ssi $P$ est pair.
}
