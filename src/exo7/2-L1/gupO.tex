\contenu{
\texte{
Donner un exemple d'application $f:{\Rr}\rightarrow {\Rr}$ non constante
telle que :
$$\forall x\in {\Rr},  f(x)=f(x^{2}). $$
On suppose $f$ continue en $0$ et en $1$, montrer que $f$ est constante.
}
