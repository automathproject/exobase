\contenu{
\texte{
Soit $p$ un nombre premier.
}
\begin{enumerate}
    \item \question{Soit $a$ une racine de $X^p-X-1\in \mathbb{F}_p[X]$, $a\in \overline{\mathbb{F} }_p$. Montrer que $a+k$ est aussi une racine de $X^p-X-1$, pour tout $k\in \mathbb{F}_p$.}
    \item \question{Montrer que $X^p-X-1$ est irréductible sur $\mathbb{F}_p$ et sur $\Z $.}
\end{enumerate}
