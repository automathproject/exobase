\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Calculs}
\module{Analyse}
\niveau{L1}
\difficulte{}

\contenu{
\texte{
Soit $f$ une fonction dérivable sur $\mathbb{R}$.
}
\begin{enumerate}
    \item \question{Calculer la dérivée de $x\mapsto \hbox{sin} (f(x)^2)$ et de
$ x \mapsto \hbox{sin} (f(x^2)).$}
    \item \question{On suppose $f(x)\not =0$ pour tout $x\in \mathbb{R}$. Calculer la
dérivée de $x\mapsto \hbox{log}(\vert f(x)\vert ).$}
\end{enumerate}
}
