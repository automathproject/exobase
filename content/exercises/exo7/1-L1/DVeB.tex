\texte{
Soit $B \in { K[X]}$ de degr{\'e} $n > 0$. On consid{\`e}re les applications :
$$\Phi : { K[X]} \to { K_{n-1}[X]}, P \mapsto R$$
et
 $$\Psi : { K[X]} \to { K[X]}, P \mapsto Q\quad
  \text{ avec }P = QB + R.$$
}
\begin{enumerate}
    \item \question{Montrer que $\Phi$ et $\Psi$ sont lin{\'e}aires. Chercher leurs noyaux et leurs images.}
    \item \question{Simplifier $\Phi(P_1P_2)$.}
\end{enumerate}
}
