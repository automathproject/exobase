\contenu{
\texte{
Soient $a,b \in \mathbb{U}$ distincts et $z \in \C$.
On note $u = \frac{z+ab\overline{z}-a-b}{a-b}$.
Montrer que $u^2 \in \R$.
}
\reponse{
$\overline{u} = -u$.
}
}
