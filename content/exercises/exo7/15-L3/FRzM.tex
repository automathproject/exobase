\texte{
Soit $H$ un sous-espace vectoriel de $\Rr^n$, $n\geq2$. Montrer que
}
\begin{enumerate}
    \item \question{si dim $H = n-1$, ${\Rr^n}\backslash H$ a deux composantes connexes;}
    \item \question{si dim $H \leq n-2$, ${\Rr^n}\backslash H$ est connexe.}
\end{enumerate}
}
